AWS CloudShell příkazy pro vytvoření a konfiguraci Cognito Identity Pool:

# Najděte User Pool ID
echo "Hledám Aeropilot User Pool..."
USER_POOL_ID=$(aws cognito-idp list-user-pools --max-results 10 --query "UserPools[?Name==AeropilotUserPool].Id" --output text)
echo "User Pool ID: $USER_POOL_ID"

# Vytvořte Identity Pool
IDENTITY_POOL=$(aws cognito-identity create-identity-pool --identity-pool-name AeropilotIdentityPool --allow-unauthenticated-identities --cognito-identity-providers ProviderName="cognito-idp.eu-central-1.amazonaws.com/$USER_POOL_ID",ClientId="b30b46cc-5882-4d11-ab16-00cc715a793d")
IDENTITY_POOL_ID=$(echo $IDENTITY_POOL | jq -r ".IdentityPoolId")
echo "Identity Pool ID: $IDENTITY_POOL_ID"

# Nastavte role pro Identity Pool
aws cognito-identity set-identity-pool-roles --identity-pool-id $IDENTITY_POOL_ID --roles "authenticated=arn:aws:iam::455982474805:role/CognitoAeropilotUserRole,unauthenticated=arn:aws:iam::455982474805:role/CognitoAeropilotGuestRole"

echo "Základní role nastaveny. Pro pokročilé role-based přiřazení použijte Cognito skupiny a pravidla v AWS Console nebo SDK."

echo "Konfigurace dokončena! Identity Pool ID: $IDENTITY_POOL_ID"
